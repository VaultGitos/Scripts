--Arsenal Wally Hub

local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/themessx/Wally-UI-Vztlx-Edition/master/VzltexArchUI.txt", true))()
local main = library:CreateWindow({text = "Byte Hub"})
local visual = library:CreateWindow({text = "Visual"})
local credits = library:CreateWindow({text='Credits'})
credits:AddLabel("Credits\nInori: Dragging\nwally: UI\nVzltex : Script Hub")

main:AddBox("WalkSpeed", function(object, focus)
    if focus then
        game:GetService('Players').LocalPlayer.Character.Humanoid.WalkSpeed = tonumber(object.Text) or 16
    end
end)

main:AddButton("NoClip Key E", function()
    noclip = false
    game:GetService('RunService').Stepped:connect(function()
    if noclip then
    game.Players.LocalPlayer.Character.Humanoid:ChangeState(11)
    end
    end)
    plr = game.Players.LocalPlayer
    mouse = plr:GetMouse()
    mouse.KeyDown:connect(function(key)
    
    if key == "e" then
    noclip = not noclip
    game.Players.LocalPlayer.Character.Humanoid:ChangeState(11)
    end
    end)
end)


visual:AddButton("Player ESP", function()
    loadstring(game:HttpGet("https://raw.githubusercontent.com/themessx/Scripts/master/ESP.txt", true))()
end)

visual:AddButton("Piggy Bot ESP", function()
    local PiggyNpc = game:GetService("Workspace").PeppaNPC.PiggyBot

    local BillboardGui = Instance.new("BillboardGui")
    local Frame = Instance.new("Frame")
    
    --Properties:
    
    BillboardGui.Parent = PiggyNpc.HumanoidRootPart
    BillboardGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    BillboardGui.Active = true
    BillboardGui.AlwaysOnTop = true
    BillboardGui.LightInfluence = 1.000
    BillboardGui.Size = UDim2.new(0, 50, 0, 50)
    
    Frame.Parent = BillboardGui
    Frame.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    Frame.BackgroundTransparency = 0.200
    Frame.Size = UDim2.new(0, 50, 0, 50)
end)
