local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/themessx/Wally-UI-Vztlx-Edition/master/VzltexArchUI.txt", true))()
local main = library:CreateWindow({text = "Vzltex Hub"})
local visual = library:CreateWindow({text = "Visual"})
local credits = library:CreateWindow({text='Credits'})
credits:AddLabel("Credits\nInori: Dragging\nwally: UI\nVzltex : Script Hub")

main:AddButton("Murder Teleport", function()
    for a,b in pairs(game:GetService("Players"):GetPlayers()) do
        for c,d in pairs(b.Backpack:GetDescendants()) do
        if d.Name == "Knife" then
        game:GetService("Players").LocalPlayer.Character:MoveTo(b.Character.HumanoidRootPart.Position+Vector3.new(0,0,2))
        end
        end
        for e,f in pairs(b.Character:GetDescendants()) do
        if f.Name == "Knife" then
        game:GetService("Players").LocalPlayer.Character:MoveTo(b.Character.HumanoidRootPart.Position+Vector3.new(0,0,2))
        end
        end
        end
end)

main:AddButton("Sheriff Teleport", function()
    for a,b in pairs(game:GetService("Players"):GetPlayers()) do
        for c,d in pairs(b.Backpack:GetDescendants()) do
        if d.Name == "Revolver" then
        game:GetService("Players").LocalPlayer.Character:MoveTo(b.Character.HumanoidRootPart.Position)
        end
        end
        for e,f in pairs(b.Character:GetDescendants()) do
        if f.Name == "Revolver" then
        game:GetService("Players").LocalPlayer.Character:MoveTo(b.Character.HumanoidRootPart.Position)
        end
        end
        end
end)

main:AddButton("Redeem All Codes", function()
    for a,b in pairs(game:GetService("ReplicatedStorage").Codes:GetChildren()) do
        game:GetService("ReplicatedStorage").Interactions.Server.RedeemCode:InvokeServer(b.Name)
        end
end)

main:AddButton("Coin Teleport", function()
    for a,b in pairs(workspace:GetChildren()) do
        if b.Name == "Coin" then
        b.Position = game:GetService("Players").LocalPlayer.Character.HumanoidRootPart.Position
        end
        end
end)

main:AddButton("Revoler Teleport", function()
    Main:CreateButton("Goto",function()
        game:GetService("Players").LocalPlayer.Character:MoveTo(game:GetService("Workspace").RevolverPickUp.Revolver.Position)
    end)
end)

main:AddButton("NoClip Key E", function()
    noclip = false
    game:GetService('RunService').Stepped:connect(function()
    if noclip then
    game.Players.LocalPlayer.Character.Humanoid:ChangeState(11)
    end
    end)
    plr = game.Players.LocalPlayer
    mouse = plr:GetMouse()
    mouse.KeyDown:connect(function(key)
    
    if key == "e" then
    noclip = not noclip
    game.Players.LocalPlayer.Character.Humanoid:ChangeState(11)
    end
    end)
end)


visual:AddButton("ESP", function()
    loadstring(game:HttpGet("https://raw.githubusercontent.com/themessx/Scripts/master/ESP.txt", true))()
end)

visual:AddButton("Murder And Sheriff ESP", function()
    playas = game.Players:GetDescendants()
    local murderer = nil
    local sheriff = nil
    
    for i,v in ipairs(playas) do
    if v:IsA("Player") then
    find = v.Character:FindFirstChild("Knife")
    find2 = v.Character:FindFirstChild("Gun")
    
    if find then
    murderer = v.Name
    end
    
    if find2 then
    sheriff = v.Name
    end
    end
    
    if v.Name == "Knife" then
    murderer = v.Parent.Parent.Name 
    end
    
    if v.Name == "Revolver" then
    sheriff = v.Parent.Parent.Name 
    end
    
    end
    
    if murderer then
    
    print("Murderer: " .. murderer)
    
    local tag = Instance.new("BillboardGui")
    tag.Parent = game.Players[murderer].Character.Head
    tag.Adornee = game.Players[murderer].Character.Head
    tag.StudsOffset = Vector3.new(0,3,0)
    tag.Size = UDim2.new(0,150,0,50)
    tag.AlwaysOnTop = true
    
    local title = Instance.new("TextLabel")
    title.Parent = tag
    title.Text = "Murderer"
    title.Size = UDim2.new(0,150,0,50)
    title.BackgroundColor3 = Color3.new(163,0,0)
    title.TextColor3 = Color3.new(255,255,255)
    title.TextScaled = true
    
    end
    
    if sheriff then
    
    print("Sheriff: " .. sheriff)
    local tag = Instance.new("BillboardGui")
    tag.Parent = game.Players[sheriff].Character.Head
    tag.Adornee = game.Players[sheriff].Character.Head
    tag.StudsOffset = Vector3.new(0,3,0)
    tag.Size = UDim2.new(0,150,0,50)
    tag.AlwaysOnTop = true
    
    local title = Instance.new("TextLabel")
    title.Parent = tag
    title.Text = "Sheriff"
    title.Size = UDim2.new(0,150,0,50)
    title.BackgroundColor3 = Color3.new(0,0,163)
    title.TextColor3 = Color3.new(255,255,255)
    title.TextScaled = true
    
    else
    print("Sheriff gun is dropped")
end
end)
