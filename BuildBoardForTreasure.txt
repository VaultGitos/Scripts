main:AddButton("Speed Bypass", function()
    while true do
        wait()
        local char = game.Players.LocalPlayer.Character
        if char:FindFirstChild("Humanoid") ~= nil then
            char.Humanoid.WalkSpeed = 100
        else
            print("Error Bypass")
        end
    end
end)

main:AddButton("JumpPower Bypass", function()
