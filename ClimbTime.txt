--Arsenal Wally Hub

local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/themessx/Wally-UI-Vztlx-Edition/master/VzltexArchUI.txt", true))()
local main = library:CreateWindow({text = "Vzltex Hub"})
local credits = library:CreateWindow({text='Credits'})
credits:AddLabel("Credits\nInori: Dragging\nwally: UI\nVzltex : Script Hub")

main:AddButton("Auto Player", function()
    local stages = {localPlayer = game:GetService("Players").LocalPlayer};

    function stages:new()
        local self = setmetatable({
            map = workspace.Map,
            currentStage = 1,
            rootPart = self.localPlayer.Character.HumanoidRootPart,
            humanoid = self.localPlayer.Character.Humanoid
        }, {__index = stages});
    
        return self;
    end;
    
    function stages:GetStages()
        local stages = {};
        local insert = table.insert;
    
        for i, v in next, self.map:GetChildren() do
            local stage = tonumber(v.Name);
            if stage then
                insert(stages, v);
            elseif v.Name == "End" then
                insert(stages, v);
            end;
        end;
    
        return stages;
    end;
    
    function stages:NextStage()
        self.rootPart.Anchored = false;
    
        for i, v in next, self:GetStages() do
            local stage = tonumber(v.Name);
            if stage and stage == self.currentStage + 1 then
                self.rootPart.CFrame = CFrame.new(v.start.Position + Vector3.new(0, 5, 0));
                self.localPlayer.Character.Humanoid.Jump = true;
                self.currentStage = self.currentStage + 1;
                repeat wait(0.2); until tostring(self.humanoid.FloorMaterial) == "Enum.Material.SmoothPlastic";
                self.rootPart.Anchored = true;
                return;
            elseif v.Name == "End" then
                self.rootPart.CFrame = CFrame.new(v.stop.Position + Vector3.new(0, 5, 0));
                self.localPlayer.Character.Humanoid.Jump = true;
                self.currentStage = self.currentStage + 1;
                repeat wait(0.2); until tostring(self.humanoid.FloorMaterial) == "Enum.Material.SmoothPlastic";
                self.rootPart.Anchored = true;
                wait(0.8);
                self.rootPart.Anchored = false;
                return;
            end;
        end;
    end;
    
    local currentStages = stages:new();
    for i = 1, #currentStages:GetStages() do
        currentStages:NextStage();
        wait(6);
    end;
end)
